by_day <- group_by(flights,year,month,day) 
  summarise(by_day, delay = mean(dep_delay, na.rm = TRUE)) %>% 
  ungroup() %>% 
  mutate(day_num = seq_along(delay)) %>%
  ggplot(aes(x = day_num, y = delay)) + geom_point() + 
  geom_smooth(method = loess, formula = y~x)
